import requests

API_KEY = "YOUR_API_KEY"
MODEL = ""

response = requests.post(
    "https://api.yourdomain.com/v1/chat/completions",
    headers={
        "Authorization": f"Bearer {API_KEY}",
        "Content-Type": "application/json"
    },
    json={
        "model": MODEL,
        "messages": [
            {"role": "user", "content": "Hello world"}
        ],
        "temperature": 0.7
    }
)

print(response.json())
